void login(user* user_head)
{

	if (power)
	{
		cout << "用户已登录！\n";
		system("pause");
		return;
	}
	user *p = user_head->next1;
	if (p == NULL)
	{
		cout << "无用户数据！\n";
		system("pause");
		return;
	}
	int judge = 0;
	char input_account[50], input_password[50];
	cout << "请输入账户ID(输入0则返回菜单)：";
	cin >> input_account;
	while (strcmp(input_account, "0") != 0)  //如果没输入0，则循环
	{
		while (p != NULL)
		{
			if (strcmp(input_account, p->getaccount()) == 0)
			{
				cout << "请输入账户密码：";
				cin >> input_password;
				if (strcmp(input_password, p->getpassword()) == 0)
				{
					operate_user = p;
					power = 1;
					judge = 1;
					cout << "登录成功！（按任意键返回主菜单）";
					system("pause");
					break;
				}
				else
				{
					while (strcmp(input_password, p->getpassword()) != 0)
					{
						cout << "密码错误，请重新输入密码:";
						cin >> input_password;
					}
					operate_user = p;
					power = 1;
					judge = 1;
					cout << "登录成功！（按任意键返回主菜单）";
					system("pause");
					break;
				}
			}
			else
			{
				p = p->next1;
			}
		}
		if (judge == 0)
		{
			cout << "该账户不存在，请重新输入(输入0返回主菜单)";
			cin >> input_account;
		}
		else break;
	}
}

user* read()
{
	fstream in;
	char accountp[50], passwordp[50];
	in.open("date.txt", ios::in);
	if (!in)
	{
		cout << "Can not open date.txt" << endl;
		exit(1);
	}
	user *p, *p1 = NULL, *head = NULL;
	while (in >> accountp)
	{
		in >> passwordp;
		p = new user;
		p->next1 = NULL; p->next2 = NULL;
		p->setuser(accountp, passwordp);
		if (head == NULL) head = p1 = p;
		else
		{
			p1->next1 = p;
			p1 = p;
		}
		char singlep = '#';
		do
		{
			record *newrecord = recordlist();
			int a, b, c;
			in >> a >> b >> c;
			newrecord->set_year2(a, b, c);
			bool d; int e;
			in >> d >> e;
			newrecord->set_money(d, e);
			char mid[50];
			in >> mid;
			newrecord->setinformation(mid);
			if (p->next2 == NULL)
			{
				p->next2 = newrecord;
			}
			else
			{
				record *q = p1->next2;
				while (p->next2 != NULL)
				{
					q = q->next2;
				}
				newrecord->setbalance(q->getbalance());
				q->next2 = newrecord;
			}
			in >> singlep;
		} while (singlep != '@');
	}
	in.close();
	cout << "导入数据完成";
	system("pause");
	return head;
}
void write(user* head)
{
	fstream out;
	out.open("date.txt", ios::out);
	if (!out)
	{
		cout << "Can not open date.txt" << endl;
		exit(1);
	}
	record* p = NULL;
	user* q = head->next1;
	while (q != NULL)
	{
		out << q->getaccount() << endl;
		out << q->getpassword() << endl;
		p = q->next2;
		while (p != NULL)
		{
			out << p->getyear() << " ";
			out << p->getmonth() << " ";
			out << p->getday() << " ";
			out << p->getsingle() << " ";
			if (p->getsingle() == 0) out << p->getoutput() << " ";
			else out << p->getincome() << " ";
			out << p->getinformation() << " ";
			p = p->next2;
			if (p == NULL)
			{
				out << "@" << endl;
			}
			else
			{
				out << "#" << endl;
			}
		}
		q = q->next1;
	}
	out.close();
}

bool time::checkTime() //时间错误检查
{
	if (month>12 || month<1)
	{
		cout << "\n\t月份输入错误!\n";
		system("pause");
		return 0;
	}
	else
	{
		if (month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12)
		{
			if (day>31 || day<1)
			{
				cout << "\n\t日期输入错误!\n";
				system("pause");
				return 0;
			}
		}
		else
		{
			if (month == 4 || month == 6 || month == 9 || month == 11)
				if (day<1 || day>30)
				{
				cout << "\n\t日期输入错误!\n";
				system("pause");
				return 0;
				}
		}
	}
	if (month == 2)
		if (day>29 || day<1)
		{
		cout << "\n\t日期输入错误!\n";
		system("pause");
		return 0;
		}
	return 1;
}
